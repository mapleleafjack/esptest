#define TFT_RST 16   // Reset pin for TFT (can be connected to ESP32-CAM reset)
#define TFT_DC 2     // Data/command pin for TFT
#define TFT_MOSI 15  // Data out (MOSI) pin for SPI
#define TFT_SCLK 14  // Clock pin for SPI
